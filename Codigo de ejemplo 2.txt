def Factorial() {
    LeeTextBox(t1, input);
    Cast(input, obj);

    if (obj >= 0) {
        res := 1;
        for (i := 1:obj, step 1) {
            res := res * i;
        }
    } else {
        res := -1;
    }

    Cast(res, output);
    ImprimeTextBox(t2, output);
}
